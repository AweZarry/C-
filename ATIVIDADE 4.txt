using System;

class Atividade
{
    static void Main(string[] args)
    {
        int num;

        Console.WriteLine("Digite sua idade:");
        num = int.Parse(Console.ReadLine());

        if (num >= 20 && num <= 30)
        {
            Console.WriteLine($"Sua idade {num} esta entre 20 e 30");
        }
        else
        {
            Console.WriteLine($"Sua idade {num} nÃ£o esta entre 20 e 30");
        }

    }
}
